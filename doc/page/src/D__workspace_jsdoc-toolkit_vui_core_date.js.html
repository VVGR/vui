130 	 *

132 	 * @result true
133 	 * 
134 	 * @name isLeapYear

395 	/**
396 	 * Returns a string representation of the date object according to
397 	 * Date.format. (Date.toString may be used in other places so I purposefully
398 	 * didn't overwrite it)
399 	 * 
400 	 * @example var dtm = new Date("01/12/2008"); dtm.asString();
401 	 * @result '12/01/2008' // (where Date.format == 'dd/mm/yyyy'
402 	 * 
403 	 * @name asString
404 	 * @type Date
405 	 * @cat Plugins/Methods/Date
406 	 * @author Kelvin Luck
407 	 */
408 	add("asString", function(format) {
409 		var r = format || Date.format;
410 		return r.split('yyyy').join(this.getFullYear()).split('yy').join(
411 				(this.getFullYear() + '').substring(2)).split('mmmm').join(
412 				this.getMonthName(false)).split('mmm').join(
413 				this.getMonthName(true)).split('mm').join(
414 				_zeroPad(this.getMonth() + 1)).split('dd').join(
415 				_zeroPad(this.getDate())).split('hh').join(
416 				_zeroPad(this.getHours())).split('min').join(
417 				_zeroPad(this.getMinutes())).split('ss').join(
418 				_zeroPad(this.getSeconds()));
419 	});

421 	/**
422 	 * Returns a new date object created from the passed String according to
423 	 * Date.format or false if the attempt to do this results in an invalid date
424 	 * object (We can't simple use Date.parse as it's not aware of locale and I
425 	 * chose not to overwrite it incase it's functionality is being relied on
426 	 * elsewhere)
427 	 * 
428 	 * @example var dtm = Date.fromString("12/01/2008"); dtm.toString();
429 	 * @result 'Sat Jan 12 2008 00:00:00' // (where Date.format == 'dd/mm/yyyy'
430 	 * 
431 	 * @name fromString
432 	 * @type Date
433 	 * @cat Plugins/Methods/Date
434 	 * @author Kelvin Luck
435 	 */
436 	Date.fromString = function(s, format) {
437 		var f = format || Date.format;
438 		var d = new Date('01/01/1977');
439 
440 		var mLength = 0;
441 
442 		var iM = f.indexOf('mmmm');
443 		if (iM > -1) {
444 			for ( var i = 0; i < Date.monthNames.length; i++) {
445 				var mStr = s.substr(iM, Date.monthNames[i].length);
446 				if (Date.monthNames[i] == mStr) {
447 					mLength = Date.monthNames[i].length - 4;
448 					break;
449 				}
450 			}
451 			d.setMonth(i);
452 		} else {
453 			iM = f.indexOf('mmm');
454 			if (iM > -1) {
455 				var mStr = s.substr(iM, 3);
456 				for ( var i = 0; i < Date.abbrMonthNames.length; i++) {
457 					if (Date.abbrMonthNames[i] == mStr)
458 						break;
459 				}
460 				d.setMonth(i);
461 			} else {
462 				d.setMonth(Number(s.substr(f.indexOf('mm'), 2)) - 1);
463 			}
464 		}
465 
466 		var iY = f.indexOf('yyyy');
467 
468 		if (iY > -1) {
469 			if (iM < iY) {
470 				iY += mLength;
471 			}
472 			d.setFullYear(Number(s.substr(iY, 4)));
473 		} else {
474 			if (iM < iY) {
475 				iY += mLength;
476 			}
477 			// TODO - this doesn't work very well - are there any rules for what
478 			// is meant by a two digit year?
479 			d.setFullYear(Number(Date.fullYearStart
480 					+ s.substr(f.indexOf('yy'), 2)));
481 		}
482 		var iD = f.indexOf('dd');
483 		if (iM < iD) {
484 			iD += mLength;
485 		}
486 		d.setDate(Number(s.substr(iD, 2)));
487 		if (isNaN(d.getTime())) {
488 			return false;
489 		}
490 		return d;
491 	};
