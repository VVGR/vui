1 /**
  2  * hashchange
  3  * @example $.cookie('the_cookie');
  4  * @from jQuery hashchange event - v1.3 http://benalman.com/projects/jquery-hashchange-plugin/
  5  */
  6 (function(jQuery, window, document, undefined) {
  7   
  8   // Reused string.
  9   var str_hashchange = 'hashchange',
 10     
 11     // Method / object references.
 12     doc = document,
 13     fake_onhashchange,
 14     special = $.event.special,
 15     
 16     // Does the browser support window.onhashchange? Note that IE8 running in
 17     // IE7 compatibility mode reports true for 'onhashchange' in window, even
 18     // though the event isn't supported, so also test document.documentMode.
 19     doc_mode = doc.documentMode,
 20     supports_onhashchange = 'on' + str_hashchange in window && ( doc_mode === undefined || doc_mode > 7 );
 21   
 22   // Get location.hash (or what you'd expect location.hash to be) sans any
 23   // leading #. Thanks for making this necessary, Firefox!
 24   function get_fragment( url ) {
 25     url = url || location.href;
 26     return '#' + url.replace( /^[^#]*#?(.*)$/, '$1' );
 27   };
 28   
 29   // Method: jQuery.fn.hashchange
 30   // 
 31   // Bind a handler to the window.onhashchange event or trigger all bound
 32   // window.onhashchange event handlers. This behavior is consistent with
 33   // jQuery's built-in event handlers.
 34   // 
 35   // Usage:
 36   // 
 37   // > jQuery(window).hashchange( [ handler ] );
 38   // 
 39   // Arguments:
 40   // 
 41   //  handler - (Function) Optional handler to be bound to the hashchange
 42   //    event. This is a "shortcut" for the more verbose form:
 43   //    jQuery(window).bind( 'hashchange', handler ). If handler is omitted,
 44   //    all bound window.onhashchange event handlers will be triggered. This
 45   //    is a shortcut for the more verbose
 46   //    jQuery(window).trigger( 'hashchange' ). These forms are described in
 47   //    the <hashchange event> section.
 48   // 
 49   // Returns:
 50   // 
 51   //  (jQuery) The initial jQuery collection of elements.
 52   
 53   // Allow the "shortcut" format $(elem).hashchange( fn ) for binding and
 54   // $(elem).hashchange() for triggering, like jQuery does for built-in events.
 55   $.fn[ str_hashchange ] = function( fn ) {
 56     return fn ? this.bind( str_hashchange, fn ) : this.trigger( str_hashchange );
 57   };
 58   
 59   // Property: jQuery.fn.hashchange.delay
 60   // 
 61   // The numeric interval (in milliseconds) at which the <hashchange event>
 62   // polling loop executes. Defaults to 50.
 63   
 64   // Property: jQuery.fn.hashchange.domain
 65   // 
 66   // If you're setting document.domain in your JavaScript, and you want hash
 67   // history to work in IE6/7, not only must this property be set, but you must
 68   // also set document.domain BEFORE jQuery is loaded into the page. This
 69   // property is only applicable if you are supporting IE6/7 (or IE8 operating
 70   // in "IE7 compatibility" mode).
 71   // 
 72   // In addition, the <jQuery.fn.hashchange.src> property must be set to the
 73   // path of the included "document-domain.html" file, which can be renamed or
 74   // modified if necessary (note that the document.domain specified must be the
 75   // same in both your main JavaScript as well as in this file).
 76   // 
 77   // Usage:
 78   // 
 79   // jQuery.fn.hashchange.domain = document.domain;
 80   
 81   // Property: jQuery.fn.hashchange.src
 82   // 
 83   // If, for some reason, you need to specify an Iframe src file (for example,
 84   // when setting document.domain as in <jQuery.fn.hashchange.domain>), you can
 85   // do so using this property. Note that when using this property, history
 86   // won't be recorded in IE6/7 until the Iframe src file loads. This property
 87   // is only applicable if you are supporting IE6/7 (or IE8 operating in "IE7
 88   // compatibility" mode).
 89   // 
 90   // Usage:
 91   // 
 92   // jQuery.fn.hashchange.src = 'path/to/file.html';
 93   
 94   $.fn[ str_hashchange ].delay = 50;
 95   /*
 96   $.fn[ str_hashchange ].domain = null;
 97   $.fn[ str_hashchange ].src = null;
 98   */
 99   
100   // Event: hashchange event
101   // 
102   // Fired when location.hash changes. In browsers that support it, the native
103   // HTML5 window.onhashchange event is used, otherwise a polling loop is
104   // initialized, running every <jQuery.fn.hashchange.delay> milliseconds to
105   // see if the hash has changed. In IE6/7 (and IE8 operating in "IE7
106   // compatibility" mode), a hidden Iframe is created to allow the back button
107   // and hash-based history to work.
108   // 
109   // Usage as described in <jQuery.fn.hashchange>:
110   // 
111   // > // Bind an event handler.
112   // > jQuery(window).hashchange( function(e) {
113   // >   var hash = location.hash;
114   // >   ...
115   // > });
116   // > 
117   // > // Manually trigger the event handler.
118   // > jQuery(window).hashchange();
119   // 
120   // A more verbose usage that allows for event namespacing:
121   // 
122   // > // Bind an event handler.
123   // > jQuery(window).bind( 'hashchange', function(e) {
124   // >   var hash = location.hash;
125   // >   ...
126   // > });
127   // > 
128   // > // Manually trigger the event handler.
129   // > jQuery(window).trigger( 'hashchange' );
130   // 
131   // Additional Notes:
132   // 
133   // * The polling loop and Iframe are not created until at least one handler
134   //   is actually bound to the 'hashchange' event.
135   // * If you need the bound handler(s) to execute immediately, in cases where
136   //   a location.hash exists on page load, via bookmark or page refresh for
137   //   example, use jQuery(window).hashchange() or the more verbose 
138   //   jQuery(window).trigger( 'hashchange' ).
139   // * The event can be bound before DOM ready, but since it won't be usable
140   //   before then in IE6/7 (due to the necessary Iframe), recommended usage is
141   //   to bind it inside a DOM ready handler.
142   
143   // Override existing $.event.special.hashchange methods (allowing this plugin
144   // to be defined after jQuery BBQ in BBQ's source code).
145   special[ str_hashchange ] = $.extend( special[ str_hashchange ], {
146     
147     // Called only when the first 'hashchange' event is bound to window.
148     setup: function() {
149       // If window.onhashchange is supported natively, there's nothing to do..
150       if ( supports_onhashchange ) { return false; }
151       
152       // Otherwise, we need to create our own. And we don't want to call this
153       // until the user binds to the event, just in case they never do, since it
154       // will create a polling loop and possibly even a hidden Iframe.
155       $( fake_onhashchange.start );
156     },
157     
158     // Called only when the last 'hashchange' event is unbound from window.
159     teardown: function() {
160       // If window.onhashchange is supported natively, there's nothing to do..
161       if ( supports_onhashchange ) { return false; }
162       
163       // Otherwise, we need to stop ours (if possible).
164       $( fake_onhashchange.stop );
165     }
166     
167   });
168   
169   // fake_onhashchange does all the work of triggering the window.onhashchange
170   // event for browsers that don't natively support it, including creating a
171   // polling loop to watch for hash changes and in IE 6/7 creating a hidden
172   // Iframe to enable back and forward.
173   fake_onhashchange = (function(){
174     var self = {},
175       timeout_id,
176       
177       // Remember the initial hash so it doesn't get triggered immediately.
178       last_hash = get_fragment(),
179       
180       fn_retval = function(val){ return val; },
181       history_set = fn_retval,
182       history_get = fn_retval;
183     
184     // Start the polling loop.
185     self.start = function() {
186       timeout_id || poll();
187     };
188     
189     // Stop the polling loop.
190     self.stop = function() {
191       timeout_id && clearTimeout( timeout_id );
192       timeout_id = undefined;
193     };
194     
195     // This polling loop checks every $.fn.hashchange.delay milliseconds to see
196     // if location.hash has changed, and triggers the 'hashchange' event on
197     // window when necessary.
198     function poll() {
199       var hash = get_fragment(),
200         history_hash = history_get( last_hash );
201       
202       if ( hash !== last_hash ) {
203         history_set( last_hash = hash, history_hash );
204         
205         $(window).trigger( str_hashchange );
206         
207       } else if ( history_hash !== last_hash ) {
208         location.href = location.href.replace( /#.*/, '' ) + history_hash;
209       }
210       
211       timeout_id = setTimeout( poll, $.fn[ str_hashchange ].delay );
212     };
213     
214     // vvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvv
215     // vvvvvvvvvvvvvvvvvvv REMOVE IF NOT SUPPORTING IE6/7/8 vvvvvvvvvvvvvvvvvvv
216     // vvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvvv
217     $.browser.msie && !supports_onhashchange && (function(){
218       // Not only do IE6/7 need the "magical" Iframe treatment, but so does IE8
219       // when running in "IE7 compatibility" mode.
220       
221       var iframe,
222         iframe_src;
223       
224       // When the event is bound and polling starts in IE 6/7, create a hidden
225       // Iframe for history handling.
226       self.start = function(){
227         if ( !iframe ) {
228           iframe_src = $.fn[ str_hashchange ].src;
229           iframe_src = iframe_src && iframe_src + get_fragment();
230           
231           // Create hidden Iframe. Attempt to make Iframe as hidden as possible
232           // by using techniques from http://www.paciellogroup.com/blog/?p=604.
233           iframe = $('<iframe tabindex="-1" title="empty"/>').hide()
234             
235             // When Iframe has completely loaded, initialize the history and
236             // start polling.
237             .one( 'load', function(){
238               iframe_src || history_set( get_fragment() );
239               poll();
240             })
241             
242             // Load Iframe src if specified, otherwise nothing.
243             .attr( 'src', iframe_src || 'javascript:0' )
244             
245             // Append Iframe after the end of the body to prevent unnecessary
246             // initial page scrolling (yes, this works).
247             .insertAfter( 'body' )[0].contentWindow;
248           
249           // Whenever `document.title` changes, update the Iframe's title to
250           // prettify the back/next history menu entries. Since IE sometimes
251           // errors with "Unspecified error" the very first time this is set
252           // (yes, very useful) wrap this with a try/catch block.
253           doc.onpropertychange = function(){
254             try {
255               if ( event.propertyName === 'title' ) {
256                 iframe.document.title = doc.title;
257               }
258             } catch(e) {}
259           };
260           
261         }
262       };
263       
264       // Override the "stop" method since an IE6/7 Iframe was created. Even
265       // if there are no longer any bound event handlers, the polling loop
266       // is still necessary for back/next to work at all!
267       self.stop = fn_retval;
268       
269       // Get history by looking at the hidden Iframe's location.hash.
270       history_get = function() {
271         return get_fragment( iframe.location.href );
272       };
273       
274       // Set a new history item by opening and then closing the Iframe
275       // document, *then* setting its location.hash. If document.domain has
276       // been set, update that as well.
277       history_set = function( hash, history_hash ) {
278         var iframe_doc = iframe.document,
279           domain = $.fn[ str_hashchange ].domain;
280         
281         if ( hash !== history_hash ) {
282           // Update Iframe with any initial `document.title` that might be set.
283           iframe_doc.title = doc.title;
284           
285           // Opening the Iframe's document after it has been closed is what
286           // actually adds a history entry.
287           iframe_doc.open();
288           
289           // Set document.domain for the Iframe document as well, if necessary.
290           domain && iframe_doc.write( '<script>document.domain="' + domain + '"</script>' );
291           
292           iframe_doc.close();
293           
294           // Update the Iframe's hash, for great justice.
295           iframe.location.hash = hash;
296         }
297       };
298       
299     })();
300     // ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
301     // ^^^^^^^^^^^^^^^^^^^ REMOVE IF NOT SUPPORTING IE6/7/8 ^^^^^^^^^^^^^^^^^^^
302     // ^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^^
303     
304     return self;
305   })();
306 
307 }(jQuery, window, document, undefined));
