 39 vui.slide = function(obj,o){
 40 	o= $.extend({
 41 		direction: 'left',
 42 		autoPlay:true,
 43 		time :1800,
 44 		eventType:'mouseover'
 45 	},o||{});
 46 	
 47 	var $this = obj;
 48 
 49 	var div = $('.slide-wrap',$this),//容器
 50 		ul = $('.slide',div),
 51 		li = $('li',ul),
 52 		divHeight = div.innerHeight(),//容器高
 53 		divWidth = div.innerWidth(),//容器宽
 54 		liHeight = li.innerHeight(),//元素高
 55 		liWidth = li.innerWidth(),//元素宽
 56 		liSize = li.length,//元素个数
 57 		ulHeight = liHeight*liSize,
 58 		ulWidth = liWidth*liSize,
 59 		visibleLiSize = Math.ceil(divHeight / liHeight),//容器范围内可容的元素个数
 60 		stepSize = Math.ceil ( liSize / visibleLiSize),//步进的总次数
 61 		remainLiSize ,//不整除时余下的元素
 62 		lastStep ,//最后一个元素步进的长度
 63 		step= 0, //每次步进的长度
 64 		currentIndex=0;
 65 
 66 	if (o.direction=='left'){ //left时设置
 67 		ulHeight = ulWidth;
 68 		liHeight = liWidth;
 69 		visibleLiSize = Math.ceil(divWidth / liWidth);
 70 		ul.width(ulWidth);
 71 	}
 72 
 73 	remainLiSize = visibleLiSize - (liSize % visibleLiSize) ;//不整除时余下的元素
 74 	if ((liSize % visibleLiSize)==0) remainLiSize=0;
 75 	lastStep = ulHeight + liHeight * remainLiSize;
 76 
 77 	//Css Set
 78 	div.css({position:'relative'});
 79 	ul.css({position:'absolute'});
 80 	li.css("float","left");
 81 
 82 	var slideOpt = li.slice(0,visibleLiSize);
 83 	slideOpt.css({position:'relative'});
 84 
 85 	var slideTrigger= $('.slide-trigger a',$this);//slide触发器
 86 //	if(!document.all) console.log(slideTrigger);
 87 //	slideTrigger.css({float:'left',cursor:'pointer'});
 88 	slideTrigger.eq(0).addClass('select');
 89 	
 90 	var slide = {
 91 		stop:function(){
 92 			ul.stop();
 93 			clearInterval(slideInterval);
 94 		},
 95 		run:function(){
 96 			ul.stop();
 97 			slideInterval = setInterval(slide.sum,o.time);			
 98 		},
 99 		sum:function(){
100 			currentIndex = (currentIndex >= stepSize -1 ? 0 : currentIndex+1 ); 
101 			slide.item(currentIndex);
102 		},
103 		item:function(index){
104 //			if(!document.all) console.log('slideItem: ' + index);
105 			slide.step(index);
106 			slide.focus(index);
107 		},
108 		setCss:function(css,value){
109 			 var o = {};
110 			o[css] = value;
111 			return o;
112 		},
113 		step:function(index){
114 			if (index==0) index=liSize;
115 			step = - index *(visibleLiSize*liHeight);
116 			if ( step <= -ulHeight){
117 				slideOpt.css(slide.setCss(o.direction,lastStep)); //最后一个元素步进设置
118 				ul.animate(slide.setCss(o.direction,-lastStep),0.9*o.time,function(){
119 					$(this).css(slide.setCss(o.direction,0));
120 					slideOpt.css(slide.setCss(o.direction,0));
121 				});
122 				return;
123 			}
124 			ul.animate(slide.setCss(o.direction,step),o.time);
125 		},
126 		focus:function(index){
127 			slideTrigger.removeClass('select');
128 			slideTrigger.eq(index).addClass('select')
129 		}
130 	}
131 	$.each(slideTrigger,function(j){
132 		var _this = slideTrigger.eq(j);
133 		_this[o.eventType](function(){
134 			slide.stop();
135 			slide.item(j);
136 //			if(!document.all) console.log(j);
137 		})
138 		_this.mouseout(function(){
139 			slide.run();
140 		})		 		
141 	})
142 
143 	if (o.autoPlay) slide.run();
144 
145 	function add(index){
146 		 $('h2').html(index);
147 	}
148 
149 	function slidePage(){
150 		var index=0;
151 		 $('.slide-page .slide-prev').click(function(){
152 			index = index-1;
153 			if(index<0) index=liSize-1;
154 			add(index);
155 			slide.stop();
156 			slide.item(index);	
157 			slide.run();
158 		 });
159 		 $('.slide-page .slide-next').click(function(){
160 			index = index+1;
161 			if(index>liSize-1) index=0;
162 			add(index);
163 			slide.stop();
164 			slide.item(index);	
165 			slide.run();
166 		 });
167 	}
168 //	slidePage();
169 }
