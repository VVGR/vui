1 /**
  2  * 自动页面滚动至某元素
  3  * @constructor
  4  * @author putaoshu@126.com
  5  * @date 2012-03-12
  6  * @param {Object} obj 主对象
  7  * @param {Number} time 页面滚动至某元素所需时间
  8  * @example vui.autoscroll($('#goTop'),100)
  9  */
 10 
 11 vui.autoscroll= function(obj,time){
 12 	var $this = $(obj);
 13 	if(!time) time = 500;
 14 	var $top = $this.offset().top;
 15 	$('html,body').animate({
 16 		scrollTop:$top
 17 	},time);
 18 }
